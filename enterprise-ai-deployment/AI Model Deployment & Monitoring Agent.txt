1. Model Deployment Agent
agents/model_deployment_agent.py
import joblib
from fastapi import FastAPI
import uvicorn
import pandas as pd

class ModelDeploymentAgent:

    def __init__(self, model_path):
        self.model = joblib.load(model_path)

    def predict(self, data):
        df = pd.DataFrame([data])
        prediction = self.model.predict(df)
        return prediction.tolist()


def create_api(model_path):

    agent = ModelDeploymentAgent(model_path)
    app = FastAPI()

    @app.get("/")
    def health():
        return {"status": "model running"}

    @app.post("/predict")
    def predict(data: dict):
        prediction = agent.predict(data)
        return {"prediction": prediction}

    return app


if __name__ == "__main__":
    app = create_api("models/fraud_model.pkl")
    uvicorn.run(app, host="0.0.0.0", port=8000)

2. Performance Monitoring Agent
agents/performance_monitor_agent.py
from monitoring.metrics_tracker import MetricsTracker
import pandas as pd

class PerformanceMonitorAgent:

    def __init__(self):
        self.metrics = MetricsTracker()

    def evaluate(self, predictions, actuals):

        df = pd.DataFrame({
            "pred": predictions,
            "actual": actuals
        })

        accuracy = (df["pred"] == df["actual"]).mean()

        report = {
            "accuracy": round(accuracy, 3),
            "total_predictions": len(df)
        }

        self.metrics.log_metrics(report)

        return report

3. Drift Detection Agent
agents/drift_detection_agent.py
import pandas as pd
from scipy.stats import ks_2samp

class DriftDetectionAgent:

    def detect_drift(self, baseline_data, current_data):

        drift_report = {}

        for column in baseline_data.columns:

            stat, p_value = ks_2samp(
                baseline_data[column],
                current_data[column]
            )

            drift_report[column] = {
                "p_value": p_value,
                "drift_detected": p_value < 0.05
            }

        return drift_report


This uses the Kolmogorov–Smirnov test to detect data distribution shifts.

from airflow import DAG
from airflow.operators.python import PythonOperator
from datetime import datetime

from pipelines.training_pipeline import train_model
from pipelines.deployment_pipeline import deploy_model


default_args = {
    "owner": "mlops",
    "start_date": datetime(2024,1,1)
}

dag = DAG(
    "ml_training_pipeline",
    default_args=default_args,
    schedule_interval="@daily"
)


train = PythonOperator(
    task_id="train_model",
    python_callable=train_model,
    dag=dag
)

deploy = PythonOperator(
    task_id="deploy_model",
    python_callable=deploy_model,

3️⃣ Canary Model Deployments

This prevents bad models from breaking production.

Instead of replacing the model instantly, the system tests it with a small percentage of traffic.

Traffic Split Logic
agents/canary_router_agent.py
import random
import requests

MODEL_V1 = "http://model-v1:8000/predict"
MODEL_V2 = "http://model-v2:8000/predict"

CANARY_PERCENTAGE = 0.1


def route_request(data):

    if random.random() < CANARY_PERCENTAGE:
        response = requests.post(MODEL_V2, json=data)
        model = "v2"

    else:
        response = requests.post(MODEL_V1, json=data)
        model = "v1"

    return {
        "model": model,
        "prediction": response.json()
    }

1. Model API Injection Risk

Problem:

The /predict endpoint accepts raw JSON with no validation.

Attack example:

 "transaction_amount": "DROP TABLE"
}


This can cause:

runtime crashes

undefined model behavior

memory abuse

Mitigation

Use Pydantic validation.

Secure API Version
"""
Model Deployment Agent
Secure FastAPI inference server with strict input validation.
"""

from fastapi import FastAPI
from pydantic import BaseModel, Field
import joblib
import pandas as pd

# -------------------------------
# Input Schema Validation
# -------------------------------

class Transaction(BaseModel):
    """
    Defines strict schema for incoming prediction requests.
    Prevents malformed or malicious input.
    """

    transaction_amount: float = Field(..., ge=0, le=10000)
    account_age: int = Field(..., ge=0)
    num_transactions: int = Field(..., ge=0, le=1000)
    merchant_risk: int = Field(..., ge=0, le=5)


class ModelDeploymentAgent:
    """
    Handles model loading and inference.
    """

    def __init__(self, model_path: str):
        self.model = joblib.load(model_path)

    def predict(self, data: Transaction):
        """
        Run prediction safely using validated data.
        """

        df = pd.DataFrame([data.dict()])
        prediction = self.model.predict(df)

        return prediction.tolist()


app = FastAPI()

agent = ModelDeploymentAgent("models/fraud_model.pkl")


@app.post("/predict")
def predict(transaction: Transaction):
    """
    Prediction endpoint.

    Accepts validated transaction data
    and returns fraud prediction.
    """

    result = agent.predict(transaction)

    return {"prediction": result}

"""
Streaming Inference Agent

Consumes transaction events from Kafka and
safely routes them to the model API.
"""

from kafka import KafkaConsumer
import json
import requests
from pydantic import ValidationError
from schema import Transaction

API_URL = "http://localhost:8000/predict"


def process_stream():
    """
    Continuously process incoming Kafka messages.
    """

    consumer = KafkaConsumer(
        "transactions",
        bootstrap_servers="localhost:9092",
        value_deserializer=lambda x: json.loads(x.decode("utf-8"))
    )

    for message in consumer:

        try:

            # Validate message schema
            transaction = Transaction(**message.value)

            response = requests.post(
                API_URL,
                json=transaction.dict()
            )

            print(response.json())

        except ValidationError:
            print("Rejected invalid transaction event")

"""
Canary Router Agent

Routes requests between production
and experimental models.
"""

import random
import requests

MODEL_V1 = "http://model-v1:8000/predict"
MODEL_V2 = "http://model-v2:8000/predict"

CANARY_PERCENTAGE = 0.10


def route_request(payload: dict):
    """
    Routes traffic between models.

    90% -> production model
    10% -> canary model
    """

    target = MODEL_V2 if random.random() < CANARY_PERCENTAGE else MODEL_V1

    try:

        response = requests.post(
            target,
            json=payload,
            timeout=2
        )

        return response.json()

    except requests.exceptions.RequestException:

        # fallback to stable model
        response = requests.post(MODEL_V1, json=payload)

        return response.json()
